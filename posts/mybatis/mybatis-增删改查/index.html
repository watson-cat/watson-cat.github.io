增删改查例子#

AccountMapper.xml#

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<!-- namespace是映射接口的路径 -->
<mapper namespace="com.eoezora.mybatisDemo.mapper.AccountMapper">
    <!-- 添加数据, 参数POJO -> EoeAccount -->
    <!-- 添加这两个 useGeneratedKeys="true" keyProperty="id" 属性在insert标签里可以返回主键 -->
    <insert id="addAccount" useGeneratedKeys="true" keyProperty="id"/>
    <insert id="addAccount" >
        INSERT INTO account (name, age, brithDay, location)
        <!-- 这里的名字要与POJO类力的变量名一致 -->
        VALUES (#{name},#{age},#{brithDay},#{location});
    </insert>
    <!-- 删除一条数据 -->
    <delete id="delAccouintById">
        DELETE
        FROM account
        WHERE id = #{id}
    </delete>
    <!-- 删除多条数据 -->
    <delete id="delAccouintByIds">
        DELETE
        FROM account
        WHERE id in
        <!-- 
            collection需要遍历的集合 
            item 集合里的元素 
            separator 分隔符
            open 前缀符
            close 后缀符
        -->
        <foreach collection="ids" item="id" separator="," open="(" close=")">
            #{id}
        </foreach>
    </delete>
    <!-- 查询所以数据 多个条件使用 and 分割 -->
    <select id="selectAccounts" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
        SELECT *
        FROM account;
    </select>
    <!-- 根据ID查询数据 -->
    <select id="selectAccount" parameterType="int" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
        SELECT *
        FROM account
        WHERE id = #{id};
    </select>
    <!-- 根据ID修改数据 -->
    <!-- 动态修改参考动态查询 -->
    <update id="setAccountAge">
        UPDATE account
        SET age = #{age}
        WHERE id = #{id}
    </update>
</mapper>

AccountMapper.class#

package com.eoezora.mybatisDemo.mapper;

import com.eoezora.mybatisDemo.pojo.EoeAccount;

import java.util.List;
//映射接口, 方法名与上方SQL映射xml里的ID一致
public interface AccountMapper {
    
    /**
     * <insert id="addAccount" parameterType="com.eoezora.mybatisDemo.pojo.EoeAccount">
    **/
    void addAccount(EoeAccount eoeAccount);
    /**
     * <select id="selectAccounts" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
    **/
    List<EoeAccount> selectAccounts();
    /**
     * <select id="selectAccount" parameterType="int" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
    **/
    EoeAccount selectAccount(@Param("id")int id);
    /**
     * <delete id="delAccouintById">
    **/
    void delAccouintById(@Param("id")int id);
    /**
     * <delete id="delAccouintByIds"> 传入数组批量删除多条数据
    **/
    void delAccouintById(@Param("ids") int[] id);
    
    //传递参数的三种方式
    
    /**
     * 多个单独的参数, @Param里的值与xml映射表的占位符一致,也就是与 xml 里的 #{} 里的一致 例如: #{name}
    **/
    List<EoeAccount> selectAccountByCondition(@Param("name") String name, @Param("age") intn age);
    /**
     * 如果参数可以封装在同一个类里,也可以直接传一个对象过来. xml #{} 将使用该对象参数名作为占位符
    **/
    List<EoeAccount> selectAccountByCondition(EoeAccount account);
   /**
     * HashMap key -> value 
     *  - 占位符名称 -> 数值
    **/
    List<EoeAccount> selectAccountByCondition(Map map);
    /**
     * 使用注解查询, 一般使用在简单的sql语句上, 复杂的语句还是使用xml映射
    **/
    @Select("SELECT * FROM account WHERE id = #{id}")
    EoeAccount selectAccount(int id);
}

使用接口进行增删改查#

try (SqlSession session = sqlSessionFactory.openSession()) {
    AccountMapper mapper = session.getMapper(AccountMapper.class);
    mapper.selectAccounts();
    mapper.addAccount();
    //需要注意的是 mybatis 对数据库进行增删改的操作后需要手动提交
    session.commit();
}
/**
 * 获取主键
 * 当数据添加完成后, 若insert标签有设置useGeneratedKeys="true" keyProperty="id" 属性
 * 则可以根据 以下方式获取返回的主键id
**/
EoeAccount ea = new EoeAccount("蔡某",25,"1919-08-10","荔枝星");
mapper.addAccount(ea);
System.out.println(ea.getId());

多条件动态查询#

在根据条件查询时,多个条件使用以下方式,传入的条件参数是固定的。比如下面这条是3个.

<!-- 查询所以数据 多个条件使用 and 分割 -->
<select id="selectAccounts" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
    SELECT *
    FROM account WHERE id = 1 AND name = 'xxx' AND AGE = 24;
</select>

有时候我们需要的是模糊查询, 用户给的条件数量是不固定的，有时一个有时三个。我们需要根据传来的条件参数数量来动态调整我们的SQL语句。MyBatis提供了许多标签来支持此功能: if | choose(when, otherwise) | trim(where, set) | foreach

<select id="selectAccounts" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
    SELECT *
    FROM account
    <!-- 使用where标签可以解决第一个条件不存在时的报错, 同时会帮我们去掉多余的AND -->
    <!-- 下面为例 当id小于0，条件不会被拼到sql语句里,且会帮我们去掉name前面的AND -->
    <where>
        <!-- test后面跟判断条件 -->
        <if test="id > 0">
            AND id = #{id}
        </if>
        <if test="name != null">
            AND name LIKE #{name}
        </if>
        <if test="age > 0">
            AND age = #{age}
        </if>
    </where>
</select>

单条件动态查询#

当我们需要单条件动态查询时，使用以下方法

<select id="selectAccounts" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
    SELECT *
    FROM account
    <where>
        <choose><!-- 类似switch -->
            <when test="id > 0"><!-- 类似case -->
                id = #{id}
            </when>
            <when test="name != null and name != ''">
                name like #{name}
            </when>
            <when test="age > 0">
                age = #{age}
            </when>
            <otherwise>1 = 1</otherwise><!-- 类似default 使用where标签可以去掉这个 -->
        </choose>
    </where>
</select>

参数传递#

Mybatis接口方法可以接受各种各样的参数，底层会对这些参数进行不同的封装处理

单个参数:#

  • POJO类型

  • Map集合

  • Coolection

  • List

  • Array

  • 其他类型

多个参数:#

类似EoeAccount selectAccount(@Param(“id”)int id, @Param(“name”)String name…);

参数需要有@Param注解,里面的值与xml映射文件里的占位符一致

#:D#

参数占位符#

#{}: 执行SQL时, 会将#{}占位符替换成 ? 会自动设置参数, 防止sql注入

${}: 拼SQL语句,会出现sql注入问题.

一般在传参数时使用 #{} 在对表名，列名进行动态设置时，使用${} 进行拼接.

parameterType#

用于设置传入的参数类型, 此属性可以省略

resultType#

执行SQL语句后返回的数据进行封装，此属性指定封装后的类型

xml中特殊字符处理#

在xml文件写SQL语句时难免会出现 大于小于的判断 而 < 是xml标签起始。所以需要进行处理。下面为例子

<select id="selectAccount" parameterType="int" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
    SELECT *
    FROM account
    <!-- 此处的小于号 < 会报错, 因为被xml判定成了不完成的标签 -->
    WHERE id < 5;
</select>
    
<!-- 正确的写法应该是 CDATA 或者 转义字符 -->
<select id="selectAccount" parameterType="int" resultType="com.eoezora.mybatisDemo.pojo.EoeAccount">
    SELECT *
    FROM account
    WHERE id
    <!-- 把他也输符号写在CDATA 里就不会报错了 -->
    <![CDATA[
	 <
    ]]> #{id};
</select>
